\section{Основная часть}

\begin{frame}{Теоретические основы}
    \begin{block}{Функция потерь}
        \begin{equation}
            L(w) = \frac{1}{n}\sum_{i=1}^{n}(y_i - f(x_i, w))^2
        \end{equation}
    \end{block}
    
    \vspace{0.3cm}
    
    \begin{alertblock}{Градиентный спуск}
        \begin{equation}
            w_{new} = w_{old} - \alpha \cdot \nabla L(w)
        \end{equation}
    \end{alertblock}
\end{frame}
